int power (int b, int p)
{
   int ret = 1;

   for ( int i = 1; i <= p; i++ ) ret *= b;

   return ret;
}

int binaryToDecimal (vector <int> &v)
{
   reverse (v.begin(), v.end());
   int ret = 0;

   Fors (i, v) {
       ret += power (2, i) * v [i];
   }

   return ret;
}
